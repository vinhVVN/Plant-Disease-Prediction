# HÀM ĐO LƯỜNG 3 CHỈ SỐ CỐT LÕI
def evaluate_test_metrics(model, dataloader, is_quantized=False):
    # Ép thiết bị về CPU để mô phỏng điện thoại và làm nổi bật sức mạnh Quantization
    device_eval = torch.device('cpu') 
    model = model.to(device_eval)
    model.eval()
    
    all_preds, all_labels = [], []
    inference_times = []
    
    # Warm-up CPU Cache (Chạy không tải vài lần để ổn định đồng hồ đo)
    dummy = torch.randn(1, 3, 224, 224).to(device_eval)
    with torch.no_grad():
        for _ in range(5): _ = model(dummy)

    with torch.no_grad():
        for images, labels in tqdm(dataloader, desc="Đang quét Test Set", leave=False):
            images = images.to(device_eval)
            
            # Bắt đầu bấm giờ Suy diễn
            start_time = time.perf_counter()
            outputs = model(images)
            end_time = time.perf_counter()
            
            # Tính thời gian trung bình cho MỖI BỨC ẢNH trong batch
            batch_time_ms = (end_time - start_time) * 1000
            time_per_image = batch_time_ms / images.size(0)
            inference_times.append(time_per_image)
            
            # Thu thập dự đoán
            preds = torch.argmax(outputs, dim=1)
            all_preds.extend(preds.cpu().numpy())
            all_labels.extend(labels.numpy())
            
    # Tính toán Metrics
    acc = accuracy_score(all_labels, all_preds)
    macro_f1 = f1_score(all_labels, all_preds, average='macro')
    avg_inference_time = np.mean(inference_times)
    
    return acc, macro_f1, avg_inference_time

# 2. HÀM TÍNH KHOẢNG TIN CẬY 95% BẰNG BOOTSTRAPPING
def compute_f1_ci(y_true, y_pred, n_bootstraps=1000, ci=95):
    bootstrapped_scores = []
    for _ in range(n_bootstraps):
        # Lấy mẫu có hoàn lại
        indices = resample(range(len(y_pred)))
        score = f1_score(np.array(y_true)[indices], np.array(y_pred)[indices], average='macro')
        bootstrapped_scores.append(score)
    
    alpha = (100 - ci) / 2.0
    lower_bound = np.percentile(bootstrapped_scores, alpha)
    upper_bound = np.percentile(bootstrapped_scores, 100 - alpha)
    return lower_bound, upper_bound

# 3. HÀM ĐO LƯỜNG TỔNG HỢP
def evaluate_test_metrics_with_ci(model, dataloader, is_quantized=False):
    device_eval = torch.device('cpu') 
    model = model.to(device_eval)
    model.eval()
    all_preds, all_labels, inference_times = [], [], []
    
    with torch.no_grad():
        for images, labels in tqdm(dataloader, desc="Đang quét Test Set", leave=False):
            images = images.to(device_eval)
            start_time = time.perf_counter()
            outputs = model(images)
            inference_times.append(((time.perf_counter() - start_time) * 1000) / images.size(0))
            all_preds.extend(torch.argmax(outputs, dim=1).cpu().numpy())
            all_labels.extend(labels.numpy())
            
    macro_f1 = f1_score(all_labels, all_preds, average='macro')
    # Tính 95% CI
    ci_lower, ci_upper = compute_f1_ci(all_labels, all_preds)
    avg_time = np.mean(inference_times)
    
    return macro_f1, ci_lower, ci_upper, avg_time

import torch
import torch.nn.functional as F
from PIL import Image
from torchvision import transforms



def predict_with_tta(model, image_path, class_names, device):
    model.eval()
    image = Image.open(image_path).convert('RGB')

    # Định nghĩa 4 luồng nhìn (Views) khác nhau cho cùng 1 bức ảnh
    normalize = transforms.Normalize(mean=[0.485, 0.456, 0.406], std=[0.229, 0.224, 0.225])
    
    tta_transforms = [
        # View 1: Chuẩn hóa cơ bản (Resize thẳng)
        transforms.Compose([transforms.Resize((224, 224)), transforms.ToTensor(), normalize]),
        
        # View 2: Center Crop (Cắt bỏ rìa nhiễu, tập trung vào giữa lá)
        transforms.Compose([transforms.Resize(256), transforms.CenterCrop(224), transforms.ToTensor(), normalize]),
        
        # View 3: Lật ngang (Horizontal Flip)
        transforms.Compose([transforms.Resize((224, 224)), transforms.RandomHorizontalFlip(p=1.0), transforms.ToTensor(), normalize]),
        
        # View 4: Phóng to nhẹ (Zoom in để xem kỹ đốm bệnh)
        transforms.Compose([transforms.Resize(256), transforms.RandomCrop(224), transforms.ToTensor(), normalize])
    ]

    # Thu thập xác suất (Softmax) từ tất cả các view
    all_probs = []
    with torch.no_grad():
        for t in tta_transforms:
            tensor_img = t(image).unsqueeze(0).to(device) # Thêm batch size = 1
            logits = model(tensor_img)
            probs = F.softmax(logits, dim=1)
            all_probs.append(probs)

    # Lấy trung bình cộng xác suất của 4 luồng nhìn (Ensemble Averaging)
    avg_probs = torch.mean(torch.stack(all_probs), dim=0).squeeze()
    
    # Tìm class có xác suất cao nhất
    top_prob, top_class_idx = torch.max(avg_probs, dim=0)
    
    predicted_class = class_names[top_class_idx.item()]
    confidence = top_prob.item() * 100

    return predicted_class, confidence, avg_probs
